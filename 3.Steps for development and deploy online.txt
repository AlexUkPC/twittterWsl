?Create a new project
    docker build -t rails-toolbox-twitterwsl --build-arg USER_ID=$(id -u) --build-arg GROUP_ID=$(id -g) -f Dockerfile.rails .
    docker run -it --rm -v $PWD:/opt/app rails-toolbox-twitterwsl rails new -d postgresql --skip-bundle twitterwsl

?Create the development environment
    Move docker-entrypoint.sh and wait-for inside the twitterwsl folder
        chmod +x twitterwsl/docker-entrypoint.sh #make it executable
    *Postgresql
        in database.yml modificari:
            in default:
                ...
                pool: <%= ENV.fetch("RAILS_MAX_THREADS") { 5 } %>
                host: <%= ENV.fetch("DATABASE_HOST"){ 'none'}  %>
                username: <%= ENV.fetch("POSTGRES_USER"){ 'none'}  %>
                password: <%= ENV.fetch("POSTGRES_PASSWORD"){ 'none'}  %>
                database: <%= ENV.fetch("POSTGRES_DB"){ 'none'}  %>
                timeout: 5000
            in development sa ramana doar <<: *default
            in production sa ramana doar <<: *default
    cd ..
    cd twitterwsl
    docker-compose run --rm --user "$(id -u):$(id -g)" web_twitterwsl bin/rails webpacker:install
    in config/webpacker.yml
        development:
            ...
            dev_server:
                ...
                port: 3036 #change this
                public: localhost:3036 #change this
                ...
                watch_options:
                    ...
                    poll: true
    in config/environments/development
	    comment config.action_view.annotate_rendered_view_with_filenames = true
    cd ..
    cd twitterwsl
    change in docker-compose.yml 
        services:
            web_aplicatie2:
                ports:
                - "3001:3000" #change only external port
            webpack_dev_server_aplicatie2:
                ports: 
                - "3036:3036" #change this
    docker-compose up

?Use it in development
    Generate a scaffold
        docker-compose exec --user "$(id -u):$(id -g)" web_twitterwsl bin/rails g scaffold User first_name last_name
        docker-compose exec --user "$(id -u):$(id -g)" web_twitterwsl bin/rails db:migrate
    Adding a gem
        docker-compose exec web_twitterwsl bundle install

        docker network connect <root_proxy_network_id> <container_id>

            <container_id> se obtine cu docker ps
